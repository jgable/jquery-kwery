Kwery.js

A jQuery plugin wrapper around common webSql functionality

Tired of writing a bunch of boilerplate code to just get your webSql code running? Kwery.js aims to abstract away some of the details and let you get messy with the stuff that matters.

Old and Busted

var db = openDatabase("myDb", "1.0", "myDb Description", 5 * 1024 * 1024);
db.transaction(function(tx) {
  tx.executeSql("Select * from Task", [], function(result) {

    $.each(result.rows, function() {
      // Do something with our Task
    });

  }, onError);
});

New Hotness

$.kwery("Select * From Task")
 .then(function(tasks) {

    $.each(tasks, function() {
      // Do something with our Task
    });

}, onError);

Batch it up

var newTasks = $.map(["money", "power", "ladies"], function(n, i) { 
  return [ i, "Get the " + n ]; 
});
$.kwery.batch("Insert into Task (Id, Description) Values (?, ?)", newTasks);
